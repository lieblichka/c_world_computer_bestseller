### painless operations -  Безболезненные операции


	# EXAMPLE 1:

		teeth = 4; // операция присваивания 

		teeth == 4; // операция сравнения 


	# EXAMPLE 2:

		teeth += 2; // equal for |		teeth = teeth + 2;

		teeth -= 2; // equal for |		teeth = teeth - 2;
	
	# EXAMPLE 3:

		teeth++; // увеличивает на 1 |	teeth = teeth + 1;

		teeth--; // уменьшает на 1	 |	teeth = teeth - 1;
