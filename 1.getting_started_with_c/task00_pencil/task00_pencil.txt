### QUESTION 3

	/* Подразумевается, что имя короче 20 латинских (и 10
	  кириллических) символов. */ 

	char ex[20];
	puts("Введите имя вашей девушки: ");
	scanf("%19s", ex);
	printf("Дорогая %s. \n\n\tС тобой поконченно.\n", ex);

	# SOLUTION

	- Массив с 20 элементами, 20 элемент для символа конца 
	строки
		
	- Сканируется 19 символов для ввода имени девушки,

	- После при помощи функции printf выводится полученная строка
	с использованием переносов строки и табуляцией


### QUESTION 4

	char suit = 'H';
	switch(suit)
	{
		case 'C':
			puts("Clubs(Трефы)");
			break;
		case 'D':
			puts("Diamonds(Бубны)");
			break;
		case 'H':
			puts("Hearts(Черви)");
			break;
		default:
			puts("Spades(Пики)");
	}

	# SOLUTION 

	- Здесь используется switch, подобие цикла с использованием
	кейсов, довольно удобная конструкция, суть работы понятна
	из самого синтаксиса переменная содержит символ 'H'
	если совпадает с кейсом, выполняется блок внутри кейса и
	break выключает цикл
